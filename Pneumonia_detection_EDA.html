def  extractInfo(dataFrame):
  patient_objects = dict()
  df = dataFrame.copy(True)
  for index, row in df.iterrows():
    # print(row)
    if not np.isnan(row['x']):
      pid = row['patientId']
      x1 = row['x']
      y1 = row['y']
      x2 = x1 + row['width']
      y2 = y1 + row['height']
      height = row['height']
      width = row['width']
      target = row['Target']
      lungType = row['class']
      new_box = [x1,y1,x2,y2,width,height]
      if pid in patient_objects:
        patient_obj = patient_objects[pid]
        old_box = patient_obj['boxes'][0]
        box =  [old_box,new_box]
        temp_obj = {
            'patientId': pid,
            'Target': target,
            'boxes': box,
            'Class': lungType
                    }
        patient_objects[pid] = temp_obj
      else:
        temp_obj = {
            'patientId': pid,
            'Target': target,
            'boxes': [new_box],
            'Class': lungType
                    }
        patient_objects[pid] = temp_obj
      # break
  return patient_objects
